sudo cp -rf * /mlops/download/

if sudo grep -r Conv2D *
then
fe=$(sudo grep -r Conv2D * | cut -d ":" -f 1)
sudo cp -rf $fe /mlops/download/cnn
sudo docker container run -dit -v /mlops/download/cnn:/root/ --name cnnmodel tensorflow/tensorflow:v1
elif sudo grep -r Dense * && ! sudo grep -r Conv2D *
then
fe=$(sudo grep -r Dense * && ! sudo grep -r Conv2D * | cut -d ":" -f 1)
sudo cp -rf $fe /mlops/download/ann
sudo docker container run -dit -v /mlops/download/ann:/root/ --name annmodel tensorflow/tensorflow:v1
elif sudo grep -r sklearn
then
fe=$(sudo grep -r sklearn | cut -d ":" -f 1)
sudo cp -rf $fe /mlops/download/lr
sudo docker container run -dit -v /mlops/download/lr:/root/ --name lcrmodel lr
else
echo "We don't support this model"
fi
